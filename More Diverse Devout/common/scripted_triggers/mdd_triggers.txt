country_controls_vatican = {
	p:xB040A0.controller = this
}

#country scope
is_papal_states = {
	OR = {
		c:PAP ?= this
		was_formed_from = PAP
	}
}

#country scope
has_papal_rule = {
	OR = {
		is_papal_states = yes
		country_controls_vatican = yes
	}
	has_law = law_type:law_theocracy
	country_has_state_religion = rel:catholic
}

#country scope
has_catholic_presence = {
	OR = {
		country_has_state_religion = rel:catholic
		var:minority_faith ?= rel:catholic
		religion_percent_country = {
			target = rel:catholic
			value >= 0.05
		}
	}
}

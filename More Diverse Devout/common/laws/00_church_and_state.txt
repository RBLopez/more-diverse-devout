# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law
# cultural_acceptance_rule and religious_acceptance_rule are triggers in the pop scope used to check if a pop should be accepted or discriminated against

law_state_religion = {
	group = lawgroup_church_and_state
	
	icon = "gfx/interface/icons/law_icons/state_religion.dds"
	
	progressiveness = -50
	
	on_enact = {
		recalculate_pop_ig_support = yes
		if = {
			limit = {
				country_has_state_religion = rel:atheist
			}
			set_state_religion = country_definition.religion
		}
	}
	
	religious_acceptance_rule = {
		is_state_religion = scope:country
	}
	
	modifier = {
		interest_group_ig_devout_pol_str_mult = 0.3
		state_radicals_from_sol_change_accepted_religion_mult = -0.1
		state_loyalists_from_sol_change_accepted_religion_mult = 0.1
		country_authority_add = 200
	}
	
	possible_political_movements = {
		law_freedom_of_conscience
	}

	pop_support = {
		value = 0
		# Clergy like theocracy
		add = {
			desc = "POP_CLERGYMEN"
			value = 0.5
		}
	}
	
	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_theocrat
		}
	}
}

law_freedom_of_conscience = {
	group = lawgroup_church_and_state
	
	icon = "gfx/interface/icons/law_icons/freedom_of_conscience.dds"
	
	progressiveness = 0

	disallowing_laws = {
		law_theocracy
	}
	
	on_enact = {
		recalculate_pop_ig_support = yes
		if = {
			limit = {
				country_has_state_religion = rel:atheist
			}
			set_state_religion = country_definition.religion
		}
	}
	
	religious_acceptance_rule = {
		shares_heritage_trait_with_state_religion = scope:country
	}
	
	modifier = {
		interest_group_ig_devout_pol_str_mult = 0.2
		state_radicals_from_sol_change_accepted_religion_mult = -0.06
		state_loyalists_from_sol_change_accepted_religion_mult = 0.06
		country_authority_add = 125
	}
	
	possible_political_movements = {
		law_religious_tolerance
		law_state_religion
	}
	
	pop_support = {
		value = 0

		add = {
			desc = "POP_ACADEMICS"
			# Academics generally against religious perseecution
			# update this to only fire if devout IG likes suppression
			if = {
				limit = {
					owner = { NOT = { has_law = law_type:law_total_separation } }
					owner = { NOT = { has_law = law_type:law_religious_tolerance } }
					is_pop_type = academics
				}
				value = 0.25
			}
		}
		
		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					religion_accepted = no
					religion = { shares_heritage_trait_with_state_religion = prev.owner }
					owner = { NOT = { has_law = law_type:law_total_separation } }
					owner = { NOT = { has_law = law_type:law_religious_tolerance } }
				}
				add = 0.5
				
				if = {
					limit = { 
						standard_of_living <= 10
					}
					add = 0.5
				}
			}			
		}

		# Clergy like theocracy
		add = {
			desc = "POP_CLERGYMEN"
			if = {
				limit = {
					owner = {
						NOT = { has_law = law_type:law_state_religion }
					}
					is_pop_type = clergymen
				}
				value = 0.5
			}
		}

	}
}

law_state_atheism = {
	group = lawgroup_church_and_state
	
	icon = "gfx/interface/icons/law_icons/state_atheism.dds"

	progressiveness = 0

	unlocking_technologies = {
		empiricism
	}

	disallowing_laws = {
		law_theocracy
		law_religious_administration
		law_religious_jurisprudence 
	}

	modifier = {
		#interest_group_ig_devout_pol_str_mult = -0.5
		state_conversion_mult = 0.25
		country_authority_add = 200
		country_clergymen_pol_str_mult = -0.9
	}
	
	can_enact = {
		ig:ig_devout = {
			is_in_government = no
		}
	}

	on_enact = {
		recalculate_pop_ig_support = yes
		
		if = {
			limit = {
				religion = {
					NOT = {
						has_discrimination_trait = atheism
					}
				}
			}
			set_state_religion = rel:atheist
			add_modifier = {
				name = antireligious_policies
				months = long_modifier_time
			}
			custom_tooltip = {
				text = state_atheism_conversion_tt
				every_scope_state = {
					limit = {
						is_incorporated = yes
					}
					convert_population = {
						target = rel:atheist
						value = 0.15
					}
				}
			}
		}
		
	}
	
	religious_acceptance_rule = {
		#this will include atheist/humanists, confusionism and buddhist faiths
		has_discrimination_trait = atheism
	}

	possible_political_movements = {
		law_religious_tolerance
		law_total_separation
	}

	pop_support = {
		value = 0
		# Some euphoric academics will support this
		add = {
			desc = "POP_ACADEMICS"
			if = {
				limit = {
					is_pop_type = academics
				}
				value = 0.1
			}
		}
		# Atheists will support this
		add = {
			desc = "POP_ATHEISTS"
			if = {
				limit = {
					has_pop_religion = atheist
				}
				value = 0.5
			}
		}
	}



}

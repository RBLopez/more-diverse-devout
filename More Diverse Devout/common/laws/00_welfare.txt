law_religious_charity = {
	group = lawgroup_welfare
	
	icon = "gfx/interface/icons/law_icons/religious_charity.dds"

	progressiveness = 25

	disallowing_laws = {
		law_total_separation
	}

	institution = institution_social_security
	institution_modifier = {
		state_dependent_wage_mult = 0.1
		state_welfare_payments_add = -0.2  #cancel default state action
	}
	
	possible_political_movements = {
		law_poor_laws
	}

	pop_support = {
		value = 0

		add = {
			desc = "POP_CLERGYMEN"
			if = {
				limit = {
					is_pop_type = clergymen
					owner.ig:ig_devout = {
						law_stance = {
							law = law_type:law_religious_charity
							value > neutral
						}
					}
				}
				value = 0.25
			}
		}

		add = {
			desc = "POP_POVERTY"
			if = {
				limit = {
					religion_accepted = yes
					owner.ig:ig_devout = {
						law_stance = {
							law = law_type:law_religious_charity
							value > neutral
						}
					}
				}
				# SoL 14 gives 0.1 pop support, 13 -> 0.2, 12 -> 0.3, ... 5 -> 1.0
				value = 15
				subtract = standard_of_living
				multiply = 0.1
				min = 0
				max = 1
			}			
		}
	}

	ai_will_do = {
		owner.ig:ig_devout = {
			law_stance = {
				law = law_type:law_religious_charity
				value > neutral
			}
		}
		exists = ruler
		ruler = {
			OR = {
				has_ideology = ideology:ideology_theocrat
				interest_group = ig:ig_devout
			}
		}
	}
}

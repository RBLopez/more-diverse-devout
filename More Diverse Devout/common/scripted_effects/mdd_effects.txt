#sets a global variable, can be used anywhere
effect_set_catholic_authority_global = {
	set_global_variable = {
		name = global_catholic_authority
		value = 0
	}
	every_country = {
		limit = {
			country_controls_vatican = yes
		}
		debug_log_scopes = yes
		change_global_variable = {
			name = global_catholic_authority
			add = vatican_country_catholic_authority
		}
	}
	every_country = {
		limit = {
			religion = rel:catholic
			NOT = {
				this = scope:vatican_country
			}
		}
		debug_log_scopes = yes
		change_global_variable = {
			name = global_catholic_authority
			add = catholic_authority
		}
	}
	every_country = {
		limit = {
			NOT = {
				religion = rel:catholic
			}
			NOT = {
				this = scope:vatican_country
			}
			OR = {
				is_country_type = recognized
				is_country_type = colonial
			}
		}
		random_scope_pop = {
			limit = {
				religion = rel:catholic
			}
			save_scope_as = catholic_pop
		}
		debug_log_scopes = yes
		change_global_variable = {
			name = global_catholic_authority
			add = catholic_minority_authority
		}
	}

}

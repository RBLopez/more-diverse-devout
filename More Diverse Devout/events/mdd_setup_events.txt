namespace = mdd_setup

mdd_setup.1 = {
	type = country_event
	hidden = yes

	immediate = {
		if = {
			limit = {
				exists = ig:ig_minority_faiths
			}
			effect_create_minority_ig_leader = yes
		}
	}
}

#Papal states setup
mdd_setup.2 = {
	type = country_event
	hidden = yes

	immediate = {
		ruler = {
			set_variable = mdd_is_pope
			set_variable = {
				name = mdd_papacy_length_yrs
				value = 5
			}
			set_global_variable = {
				name = mdd_current_pope
				value = this
			}
		}

		#override historical cardinals with templates that match starting history vaues
		create_character = {
			template = luigi_lambruschini_template
			ig_leader = yes
			interest_group = ig_devout
			ideology = ideology_moderate
		}
		create_character = {
			template = giovanni_ferretti_template
			ig_leader = yes
			interest_group = ig_petty_bourgeoisie
			ideology = ideology_theocrat
		}
	}
}
